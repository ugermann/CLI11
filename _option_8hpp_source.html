627  std::string get_name(bool positional = false, //<[input] Show the positional name
628  bool all_options = false //<[input] Show every option
629  ) const {
630 
631  if(all_options) {
632 
633  std::vector<std::string> name_list;
634 
636  if((positional && (!pname_.empty())) || (snames_.empty() && lnames_.empty())) {
637  name_list.push_back(pname_);
638  }
639  if((get_items_expected() == 0) && (!fnames_.empty())) {
640  for(const std::string &sname : snames_) {
641  name_list.push_back("-" + sname);
642  if(check_fname(sname)) {
643  name_list.back() += "{" + get_flag_value(sname, "") + "}";
644  }
645  }
646 
647  for(const std::string &lname : lnames_) {
648  name_list.push_back("--" + lname);
649  if(check_fname(lname)) {
650  name_list.back() += "{" + get_flag_value(lname, "") + "}";
651  }
652  }
653  } else {
654  for(const std::string &sname : snames_)
655  name_list.push_back("-" + sname);
656 
657  for(const std::string &lname : lnames_)
658  name_list.push_back("--" + lname);
659  }
660 
661  return detail::join(name_list);
662  }
663 
664  // This returns the positional name no matter what
665  if(positional)
666  return pname_;
667 
668  // Prefer long name
669  if(!lnames_.empty())
670  return std::string(2, '-') + lnames_[0];
671 
672  // Or short name if no long name
673  if(!snames_.empty())
674  return std::string(1, '-') + snames_[0];
675 
676  // If positional is the only name, it's okay to use that
677  return pname_;
678  }

685  void run_callback() {
686 
687  callback_run_ = true;
688 
689  // Run the validators (can change the string)
690  if(!validators_.empty()) {
691  int index = 0;
692  // this is not available until multi_option_policy with type_size_>0 is enabled and functional
693  // if(type_size_ > 0 && multi_option_policy_ == CLI::MultiOptionPolicy::TakeLast) {
694  // index = type_size_ - static_cast<int>(results_.size());
695  //}
696  if(type_size_ < 0 && multi_option_policy_ == CLI::MultiOptionPolicy::TakeLast) { // for vector operations
697  index = expected_ - static_cast<int>(results_.size());
698  }
699  for(std::string &result : results_) {
700  auto err_msg = _validate(result, index);
701  ++index;
702  if(!err_msg.empty())
703  throw ValidationError(get_name(), err_msg);
704  }
705  }
706  if(!(callback_)) {
707  return;
708  }
709  bool local_result;
710 
711  // Num items expected or length of vector, always at least 1
712  // Only valid for a trimming policy
713  int trim_size =
714  std::min<int>(std::max<int>(std::abs(get_items_expected()), 1), static_cast<int>(results_.size()));
715 
716  // Operation depends on the policy setting
717  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
718  // Allow multi-option sizes (including 0)
719  results_t partial_result{results_.end() - trim_size, results_.end()};
720  local_result = !callback_(partial_result);
721 
722  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
723  results_t partial_result{results_.begin(), results_.begin() + trim_size};
724  local_result = !callback_(partial_result);
725 
726  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
727  results_t partial_result = {detail::join(results_, "\n")};
728  local_result = !callback_(partial_result);
729 
730  } else {
731  // Exact number required
732  if(get_items_expected() > 0) {
733  if(results_.size() != static_cast<size_t>(get_items_expected()))
734  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
735  // Variable length list
736  } else if(get_items_expected() < 0) {
737  // Require that this be a multiple of expected size and at least as many as expected
738  if(results_.size() < static_cast<size_t>(-get_items_expected()) ||
739  results_.size() % static_cast<size_t>(std::abs(get_type_size())) != 0u)
740  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
741  }
742  local_result = !callback_(results_);
743  }
744 
745  if(local_result)
746  throw ConversionError(get_name(), results_);
747  }

807  std::string get_flag_value(std::string name, std::string input_value) const {
808  static const std::string trueString{"true"};
809  static const std::string falseString{"false"};
810  static const std::string emptyString{"{}"};
811  // check for disable flag override_
812  if(disable_flag_override_) {
813  if(!((input_value.empty()) || (input_value == emptyString))) {
814  auto default_ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
815  if(default_ind >= 0) {
816  // We can static cast this to size_t because it is more than 0 in this block
817  if(default_flag_values_[static_cast<size_t>(default_ind)].second != input_value) {
818  throw(ArgumentMismatch::FlagOverride(name));
819  }
820  } else {
821  if(input_value != trueString) {
822  throw(ArgumentMismatch::FlagOverride(name));
823  }
824  }
825  }
826  }
827  auto ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
828  if((input_value.empty()) || (input_value == emptyString)) {
829  return (ind < 0) ? trueString : default_flag_values_[static_cast<size_t>(ind)].second;
830  }
831  if(ind < 0) {
832  return input_value;
833  }
834  if(default_flag_values_[static_cast<size_t>(ind)].second == falseString) {
835  try {
836  auto val = detail::to_flag_value(input_value);
837  return (val == 1) ? falseString : (val == (-1) ? trueString : std::to_string(-val));
838  } catch(const std::invalid_argument &) {
839  return input_value;
840  }
841  } else {
842  return input_value;
843  }
844  }

873  template <typename T,
874  enable_if_t<!is_vector<T>::value && !std::is_const<T>::value, detail::enabler> = detail::dummy>
875  void results(T &output) const {
876  bool retval;
877  if(results_.empty()) {
878  retval = detail::lexical_cast(default_str_, output);
879  } else if(results_.size() == 1) {
880  retval = detail::lexical_cast(results_[0], output);
881  } else {
882  switch(multi_option_policy_) {
883  case MultiOptionPolicy::TakeFirst:
884  retval = detail::lexical_cast(results_.front(), output);
885  break;
886  case MultiOptionPolicy::TakeLast:
887  default:
888  retval = detail::lexical_cast(results_.back(), output);
889  break;
890  case MultiOptionPolicy::Throw:
891  throw ConversionError(get_name(), results_);
892  case MultiOptionPolicy::Join:

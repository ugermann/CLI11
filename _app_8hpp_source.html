466  Option *add_option(std::string option_name,
467  callback_t option_callback,
468  std::string option_description = "",
469  bool defaulted = false,
470  std::function<std::string()> func = {}) {
471  Option myopt{option_name, option_description, option_callback, this};
472 
473  if(std::find_if(std::begin(options_), std::end(options_), [&myopt](const Option_p &v) {
474  return *v == myopt;
475  }) == std::end(options_)) {
476  options_.emplace_back();
477  Option_p &option = options_.back();
478  option.reset(new Option(option_name, option_description, option_callback, this));
479 
480  // Set the default string capture function
481  option->default_function(func);
482 
483  // For compatibility with CLI11 1.7 and before, capture the default string here
484  if(defaulted)
485  option->capture_default_str();
486 
487  // Transfer defaults to the new option
488  option_defaults_.copy_to(option.get());
489 
490  // Don't bother to capture if we already did
491  if(!defaulted && option->get_always_capture_default())
492  option->capture_default_str();
493 
494  return option.get();
495  }
496  throw OptionAlreadyAdded(myopt.get_name());
497  }

502  Option *add_option(std::string option_name,
503  T &variable,
504  std::string option_description = "",
505  bool defaulted = false) {
506 
507  auto fun = [&variable](CLI::results_t res) { // comment for spacing
508  return detail::lexical_conversion<T, XC>(res, variable);
509  };
510 
511  Option *opt = add_option(option_name, fun, option_description, defaulted, [&variable]() {
512  return CLI::detail::checked_to_string<T, XC>(variable);
513  });
514  opt->type_name(detail::type_name<XC>());
515  // these must be actual variable since (std::max) sometimes is defined in terms of references and references
516  // to structs used in the evaluation can be temporary so that would cause issues.
517  auto Tcount = detail::type_count<T>::value;
518  auto XCcount = detail::type_count<XC>::value;
519  opt->type_size((std::max)(Tcount, XCcount));
520  return opt;

593  Option *_add_flag_internal(std::string flag_name, CLI::callback_t fun, std::string flag_description) {
594  Option *opt;
595  if(detail::has_default_flag_values(flag_name)) {
596  // check for default values and if it has them
597  auto flag_defaults = detail::get_default_flag_values(flag_name);
598  detail::remove_default_flag_values(flag_name);
599  opt = add_option(std::move(flag_name), std::move(fun), std::move(flag_description), false);
600  for(const auto &fname : flag_defaults)
601  opt->fnames_.push_back(fname.first);
602  opt->default_flag_values_ = std::move(flag_defaults);
603  } else {
604  opt = add_option(std::move(flag_name), std::move(fun), std::move(flag_description), false);
605  }
606  // flags cannot have positional values
607  if(opt->get_positional()) {
608  auto pos_name = opt->get_name(true);
609  remove_option(opt);
610  throw IncorrectConstruction::PositionalFlag(pos_name);
611  }
612 
613  opt->type_size(0);
614  return opt;
615  }

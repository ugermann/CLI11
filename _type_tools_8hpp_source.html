495 inline int64_t to_flag_value(std::string val) {
496  static const std::string trueString("true");
497  static const std::string falseString("false");
498  if(val == trueString) {
499  return 1;
500  }
501  if(val == falseString) {
502  return -1;
503  }
504  val = detail::to_lower(val);
505  int64_t ret;
506  if(val.size() == 1) {
507  if(val[0] >= '1' && val[0] <= '9') {
508  return (static_cast<int64_t>(val[0]) - '0');
509  }
510  switch(val[0]) {
511  case '0':
512  case 'f':
513  case 'n':
514  case '-':
515  ret = -1;
516  break;
517  case 't':
518  case 'y':
519  case '+':
520  ret = 1;
521  break;
522  default:
523  throw std::invalid_argument("unrecognized character");
524  }
525  return ret;
526  }
527  if(val == trueString || val == "on" || val == "yes" || val == "enable") {
528  ret = 1;
529  } else if(val == falseString || val == "off" || val == "no" || val == "disable") {
530  ret = -1;
531  } else {
532  ret = std::stoll(val);
533  }
534  return ret;
535 }

540  try {
541  size_t n = 0;

546  return false;
547  } catch(const std::out_of_range &) {
548  return false;

157  Validator operator&(const Validator &other) const {
158  Validator newval;
159 
160  newval._merge_description(*this, other, " AND ");
161 
162  // Give references (will make a copy in lambda function)
163  const std::function<std::string(std::string & filename)> &f1 = func_;
164  const std::function<std::string(std::string & filename)> &f2 = other.func_;
165 
166  newval.func_ = [f1, f2](std::string &input) {
167  std::string s1 = f1(input);
168  std::string s2 = f2(input);
169  if(!s1.empty() && !s2.empty())
170  return std::string("(") + s1 + ") AND (" + s2 + ")";
171  else
172  return s1 + s2;
173  };
174 
175  newval.active_ = (active_ & other.active_);
176  newval.application_index_ = application_index_;
177  return newval;
178  }
179 
182  Validator operator|(const Validator &other) const {
183  Validator newval;
184 
185  newval._merge_description(*this, other, " OR ");
186 
187  // Give references (will make a copy in lambda function)
188  const std::function<std::string(std::string &)> &f1 = func_;
189  const std::function<std::string(std::string &)> &f2 = other.func_;
190 
191  newval.func_ = [f1, f2](std::string &input) {
192  std::string s1 = f1(input);
193  std::string s2 = f2(input);
194  if(s1.empty() || s2.empty())
195  return std::string();
196 
197  return std::string("(") + s1 + ") OR (" + s2 + ")";
198  };
199  newval.active_ = (active_ & other.active_);
200  newval.application_index_ = application_index_;
201  return newval;
202  }
203 
205  Validator operator!() const {
206  Validator newval;
207  const std::function<std::string()> &dfunc1 = desc_function_;
208  newval.desc_function_ = [dfunc1]() {
209  auto str = dfunc1();
210  return (!str.empty()) ? std::string("NOT ") + str : std::string{};
211  };
212  // Give references (will make a copy in lambda function)
213  const std::function<std::string(std::string & res)> &f1 = func_;
214 
215  newval.func_ = [f1, dfunc1](std::string &test) -> std::string {
216  std::string s1 = f1(test);
217  if(s1.empty()) {
218  return std::string("check ") + dfunc1() + " succeeded improperly";
219  }
220  return std::string{};
221  };
222  newval.active_ = active_;
223  newval.application_index_ = application_index_;
224  return newval;
225  }
226 
227  private:
228  void _merge_description(const Validator &val1, const Validator &val2, const std::string &merger) {
229 
230  const std::function<std::string()> &dfunc1 = val1.desc_function_;
231  const std::function<std::string()> &dfunc2 = val2.desc_function_;
232 
233  desc_function_ = [=]() {
234  std::string f1 = dfunc1();
235  std::string f2 = dfunc2();
236  if((f1.empty()) || (f2.empty())) {
237  return f1 + f2;
238  }
239  return std::string(1, '(') + f1 + ')' + merger + '(' + f2 + ')';
240  };
241  }

515 auto search(const T &set, const V &val) -> std::pair<bool, decltype(std::begin(detail::smart_deref(set)))> {
516  using element_t = typename detail::element_type<T>::type;

519  return (detail::pair_adaptor<element_t>::first(v) == val);
520  });
521  return {(it != std::end(setref)), it};
522 }

533 template <typename T, typename V>
534 auto search(const T &set, const V &val, const std::function<V(V)> &filter_function)
535  -> std::pair<bool, decltype(std::begin(detail::smart_deref(set)))> {
536  using element_t = typename detail::element_type<T>::type;
537  // do the potentially faster first search
538  auto res = search(set, val);
539  if((res.first) || (!(filter_function))) {
540  return res;
541  }
542  // if we haven't found it do the longer linear search with all the element translations
543  auto &setref = detail::smart_deref(set);
544  auto it = std::find_if(std::begin(setref), std::end(setref), [&](decltype(*std::begin(setref)) v) {
545  V a{detail::pair_adaptor<element_t>::first(v)};
546  a = filter_function(a);
547  return (a == val);
548  });
549  return {(it != std::end(setref)), it};
550 }

556 template <typename T>
557 inline typename std::enable_if<std::is_signed<T>::value, T>::type overflowCheck(const T &a, const T &b) {
558  if((a > 0) == (b > 0)) {
559  return ((std::numeric_limits<T>::max)() / (std::abs)(a) < (std::abs)(b));
560  } else {
561  return ((std::numeric_limits<T>::min)() / (std::abs)(a) > -(std::abs)(b));
562  }
563 }

567  return ((std::numeric_limits<T>::max)() / a < b);

571 template <typename T> typename std::enable_if<std::is_integral<T>::value, bool>::type checked_multiply(T &a, T b) {
572  if(a == 0 || b == 0 || a == 1 || b == 1) {
573  a *= b;
574  return true;
575  }
576  if(a == (std::numeric_limits<T>::min)() || b == (std::numeric_limits<T>::min)()) {
577  return false;
578  }
579  if(overflowCheck(a, b)) {
580  return false;
581  }
582  a *= b;
583  return true;
584 }

589  T c = a * b;

599 class IsMember : public Validator {
600  public:
601  using filter_fn_t = std::function<std::string(std::string)>;
602 
604  template <typename T, typename... Args>
605  explicit IsMember(std::initializer_list<T> values, Args &&... args)
606  : IsMember(std::vector<T>(values), std::forward<Args>(args)...) {}
607 
609  template <typename T> explicit IsMember(T &&set) : IsMember(std::forward<T>(set), nullptr) {}
610 
613  template <typename T, typename F> explicit IsMember(T set, F filter_function) {
614 
615  // Get the type of the contained item - requires a container have ::value_type
616  // if the type does not have first_type and second_type, these are both value_type
617  using element_t = typename detail::element_type<T>::type; // Removes (smart) pointers if needed
618  using item_t = typename detail::pair_adaptor<element_t>::first_type; // Is value_type if not a map
619 
620  using local_item_t = typename IsMemberType<item_t>::type; // This will convert bad types to good ones
621  // (const char * to std::string)
622 
623  // Make a local copy of the filter function, using a std::function if not one already
624  std::function<local_item_t(local_item_t)> filter_fn = filter_function;
625 
626  // This is the type name for help, it will take the current version of the set contents
627  desc_function_ = [set]() { return detail::generate_set(detail::smart_deref(set)); };
628 
629  // This is the function that validates
630  // It stores a copy of the set pointer-like, so shared_ptr will stay alive
631  func_ = [set, filter_fn](std::string &input) {
632  local_item_t b;
633  if(!detail::lexical_cast(input, b)) {
634  throw ValidationError(input); // name is added later
635  }
636  if(filter_fn) {
637  b = filter_fn(b);
638  }
639  auto res = detail::search(set, b, filter_fn);
640  if(res.first) {
641  // Make sure the version in the input string is identical to the one in the set
642  if(filter_fn) {
643  input = detail::to_string(detail::pair_adaptor<element_t>::first(*(res.second)));
644  }
645 
646  // Return empty error string (success)
647  return std::string{};
648  }
649 
650  // If you reach this point, the result was not found
651  std::string out(" not in ");
652  out += detail::generate_set(detail::smart_deref(set));
653  return out;
654  };
655  }
656 
658  template <typename T, typename... Args>
659  IsMember(T &&set, filter_fn_t filter_fn_1, filter_fn_t filter_fn_2, Args &&... other)
660  : IsMember(std::forward<T>(set),
661  [filter_fn_1, filter_fn_2](std::string a) { return filter_fn_2(filter_fn_1(a)); },
662  other...) {}
663 };

669 class Transformer : public Validator {
670  public:
671  using filter_fn_t = std::function<std::string(std::string)>;
672 
674  template <typename... Args>
675  explicit Transformer(std::initializer_list<std::pair<std::string, std::string>> values, Args &&... args)
676  : Transformer(TransformPairs<std::string>(values), std::forward<Args>(args)...) {}

679  template <typename T> explicit Transformer(T &&mapping) : Transformer(std::forward<T>(mapping), nullptr) {}
680 
683  template <typename T, typename F> explicit Transformer(T mapping, F filter_function) {
684 
685  static_assert(detail::pair_adaptor<typename detail::element_type<T>::type>::value,
686  "mapping must produce value pairs");
687  // Get the type of the contained item - requires a container have ::value_type
688  // if the type does not have first_type and second_type, these are both value_type
689  using element_t = typename detail::element_type<T>::type; // Removes (smart) pointers if needed
690  using item_t = typename detail::pair_adaptor<element_t>::first_type; // Is value_type if not a map
691  using local_item_t = typename IsMemberType<item_t>::type; // This will convert bad types to good ones
692  // (const char * to std::string)
693 
694  // Make a local copy of the filter function, using a std::function if not one already
695  std::function<local_item_t(local_item_t)> filter_fn = filter_function;
696 
697  // This is the type name for help, it will take the current version of the set contents
698  desc_function_ = [mapping]() { return detail::generate_map(detail::smart_deref(mapping)); };
699 
700  func_ = [mapping, filter_fn](std::string &input) {
701  local_item_t b;
702  if(!detail::lexical_cast(input, b)) {
703  return std::string();
704  // there is no possible way we can match anything in the mapping if we can't convert so just return
705  }
706  if(filter_fn) {
707  b = filter_fn(b);
708  }
709  auto res = detail::search(mapping, b, filter_fn);
710  if(res.first) {
711  input = detail::to_string(detail::pair_adaptor<element_t>::second(*res.second));
712  }
713  return std::string{};
714  };
715  }

740  template <typename T, typename F> explicit CheckedTransformer(T mapping, F filter_function) {
741 
742  static_assert(detail::pair_adaptor<typename detail::element_type<T>::type>::value,
743  "mapping must produce value pairs");
744  // Get the type of the contained item - requires a container have ::value_type
745  // if the type does not have first_type and second_type, these are both value_type
746  using element_t = typename detail::element_type<T>::type; // Removes (smart) pointers if needed
747  using item_t = typename detail::pair_adaptor<element_t>::first_type; // Is value_type if not a map
748  using local_item_t = typename IsMemberType<item_t>::type; // This will convert bad types to good ones
749  // (const char * to std::string)
750  using iteration_type_t = typename detail::pair_adaptor<element_t>::value_type; // the type of the object pair //
751  // the type of the object pair
752 
753  // Make a local copy of the filter function, using a std::function if not one already
754  std::function<local_item_t(local_item_t)> filter_fn = filter_function;
755 
756  auto tfunc = [mapping]() {
757  std::string out("value in ");
758  out += detail::generate_map(detail::smart_deref(mapping)) + " OR {";
759  out += detail::join(
760  detail::smart_deref(mapping),
761  [](const iteration_type_t &v) { return detail::to_string(detail::pair_adaptor<element_t>::second(v)); },
762  ",");
763  out.push_back('}');
764  return out;
765  };
766 
767  desc_function_ = tfunc;
768 
769  func_ = [mapping, tfunc, filter_fn](std::string &input) {
770  local_item_t b;
771  bool converted = detail::lexical_cast(input, b);
772  if(converted) {
773  if(filter_fn) {
774  b = filter_fn(b);
775  }
776  auto res = detail::search(mapping, b, filter_fn);
777  if(res.first) {
778  input = detail::to_string(detail::pair_adaptor<element_t>::second(*res.second));
779  return std::string{};
780  }
781  }
782  for(const auto &v : detail::smart_deref(mapping)) {
783  auto output_string = detail::to_string(detail::pair_adaptor<element_t>::second(v));
784  if(output_string == input) {
785  return std::string();
786  }
787  }
788 
789  return "Check " + input + " " + tfunc() + " FAILED";
790  };
791  }

825 class AsNumberWithUnit : public Validator {
826  public:
831  enum Options {
832  CASE_SENSITIVE = 0,
833  CASE_INSENSITIVE = 1,
834  UNIT_OPTIONAL = 0,
835  UNIT_REQUIRED = 2,
836  DEFAULT = CASE_INSENSITIVE | UNIT_OPTIONAL
837  };
838 
839  template <typename Number>
840  explicit AsNumberWithUnit(std::map<std::string, Number> mapping,
841  Options opts = DEFAULT,
842  const std::string &unit_name = "UNIT") {
843  description(generate_description<Number>(unit_name, opts));
844  validate_mapping(mapping, opts);
845 
846  // transform function
847  func_ = [mapping, opts](std::string &input) -> std::string {
848  Number num;
849 
850  detail::rtrim(input);
851  if(input.empty()) {
852  throw ValidationError("Input is empty");
853  }
854 
855  // Find split position between number and prefix
856  auto unit_begin = input.end();
857  while(unit_begin > input.begin() && std::isalpha(*(unit_begin - 1), std::locale())) {
858  --unit_begin;
859  }
860 
861  std::string unit{unit_begin, input.end()};
862  input.resize(static_cast<size_t>(std::distance(input.begin(), unit_begin)));
863  detail::trim(input);
864 
865  if(opts & UNIT_REQUIRED && unit.empty()) {
866  throw ValidationError("Missing mandatory unit");
867  }
868  if(opts & CASE_INSENSITIVE) {
869  unit = detail::to_lower(unit);
870  }
871 
872  bool converted = detail::lexical_cast(input, num);
873  if(!converted) {
874  throw ValidationError("Value " + input + " could not be converted to " + detail::type_name<Number>());
875  }
876 
877  if(unit.empty()) {
878  // No need to modify input if no unit passed
879  return {};
880  }
881 
882  // find corresponding factor
883  auto it = mapping.find(unit);
884  if(it == mapping.end()) {
885  throw ValidationError(unit +
886  " unit not recognized. "
887  "Allowed values: " +
888  detail::generate_map(mapping, true));
889  }
890 
891  // perform safe multiplication
892  bool ok = detail::checked_multiply(num, it->second);
893  if(!ok) {
894  throw ValidationError(detail::to_string(num) + " multiplied by " + unit +
895  " factor would cause number overflow. Use smaller value.");
896  }
897  input = detail::to_string(num);
898 
899  return {};
900  };
901  }
902 
903  private:
906  template <typename Number> static void validate_mapping(std::map<std::string, Number> &mapping, Options opts) {
907  for(auto &kv : mapping) {
908  if(kv.first.empty()) {
909  throw ValidationError("Unit must not be empty.");
910  }
911  if(!detail::isalpha(kv.first)) {
912  throw ValidationError("Unit must contain only letters.");
913  }
914  }
915 
916  // make all units lowercase if CASE_INSENSITIVE
917  if(opts & CASE_INSENSITIVE) {
918  std::map<std::string, Number> lower_mapping;
919  for(auto &kv : mapping) {
920  auto s = detail::to_lower(kv.first);
921  if(lower_mapping.count(s)) {
922  throw ValidationError("Several matching lowercase unit representations are found: " + s);
923  }
924  lower_mapping[detail::to_lower(kv.first)] = kv.second;
925  }
926  mapping = std::move(lower_mapping);
927  }
928  }
929 
931  template <typename Number> static std::string generate_description(const std::string &name, Options opts) {
932  std::stringstream out;
933  out << detail::type_name<Number>() << ' ';
934  if(opts & UNIT_REQUIRED) {
935  out << name;
936  } else {
937  out << '[' << name << ']';
938  }
939  return out.str();
940  }

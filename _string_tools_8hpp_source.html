215  auto loc = flags.find_first_of('{');
216  while(loc != std::string::npos) {
217  auto finish = flags.find_first_of("},", loc + 1);
218  if((finish != std::string::npos) && (flags[finish] == '}')) {

269 inline std::vector<std::string> split_up(std::string str) {
270 
271  const std::string delims("\'\"`");
272  auto find_ws = [](char ch) { return std::isspace<char>(ch, std::locale()); };
273  trim(str);
274 
275  std::vector<std::string> output;
276  bool embeddedQuote = false;
277  char keyChar = ' ';
278  while(!str.empty()) {
279  if(delims.find_first_of(str[0]) != std::string::npos) {
280  keyChar = str[0];
281  auto end = str.find_first_of(keyChar, 1);
282  while((end != std::string::npos) && (str[end - 1] == '\\')) { // deal with escaped quotes
283  end = str.find_first_of(keyChar, end + 1);
284  embeddedQuote = true;
285  }
286  if(end != std::string::npos) {
287  output.push_back(str.substr(1, end - 1));
288  str = str.substr(end + 1);
289  } else {
290  output.push_back(str.substr(1));
291  str = "";
292  }
293  } else {
294  auto it = std::find_if(std::begin(str), std::end(str), find_ws);
295  if(it != std::end(str)) {
296  std::string value = std::string(str.begin(), it);
297  output.push_back(value);
298  str = std::string(it, str.end());
299  } else {
300  output.push_back(str);
301  str = "";
302  }
303  }
304  // transform any embedded quotes into the regular character
305  if(embeddedQuote) {
306  output.back() = find_and_replace(output.back(), std::string("\\") + keyChar, std::string(1, keyChar));
307  embeddedQuote = false;
308  }
309  trim(str);
310  }
311  return output;
312 }

334 inline size_t escape_detect(std::string &str, size_t offset) {
335  auto next = str[offset + 1];
336  if((next == '\"') || (next == '\'') || (next == '`')) {
337  auto astart = str.find_last_of("-/ \"\'`", offset - 1);
338  if(astart != std::string::npos) {
339  if(str[astart] == ((str[offset] == '=') ? '-' : '/'))
340  str[offset] = ' '; // interpret this as a space so the split_up works properly
341  }
342  }
343  return offset + 1;
344 }
